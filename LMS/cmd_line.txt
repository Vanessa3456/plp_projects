WINDOWS
1.Create file
-use echo = echo "This is a Word Doc" >my_doc.docx
2.Rename file
use ren
ren my_doc.docx new_doc.docx
3.Copying file
use copy
copy new_doc.docx backup_doc.docx
4.Moving file
mov backup_doc.docx C:\Documents\Moved_Document.docx
